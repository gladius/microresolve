MicroResolve ships a Node.js package with pre-built native binaries. No build step, no native compiler needed.

## Install

```bash
npm install microresolve
# or
yarn add microresolve
# or
pnpm add microresolve
```

Pre-built binaries for Linux, macOS, and Windows (x86_64 and arm64).

## Basic usage

```typescript
import { MicroResolve } from 'microresolve';

const engine = new MicroResolve();
const ns = engine.namespace('support');

ns.addIntent('cancel_order', [
  'cancel my order',
  'I want to cancel',
  'stop my order',
]);
ns.addIntent('track_order', [
  'where is my package',
  'track my order',
  'shipping status',
]);

const result = ns.resolve('I need to cancel');
for (const m of result.intents) {
  console.log(`${m.id}: ${m.score.toFixed(2)} [${m.band}]`);
}
// cancel_order: 0.91 [High]
```

## CommonJS

```javascript
const { MicroResolve } = require('microresolve');
const engine = new MicroResolve();
```

## Persist to disk

```typescript
import { MicroResolve } from 'microresolve';

const engine = new MicroResolve({ dataDir: '~/.local/share/microresolve' });
const ns = engine.namespace('support');
// ... add intents and phrases ...
await engine.flush();
```

## Custom threshold per namespace

```typescript
const security = engine.namespace('security');
security.updateNamespace({ defaultThreshold: 1.3 });
```

## Acting on disposition

```typescript
const result = ns.resolve('cancel my order and get a refund');
if (result.disposition === 'Confident') {
  handleIntent(result.intents[0].id);
} else if (result.disposition === 'LowConfidence') {
  askUserOrRouteToLlm(result.intents);
} else {
  fallback();
}
```

## Add phrases at runtime

```typescript
const result = ns.addPhrase('cancel_order', 'abort my order');
console.log(result.added);      // true
console.log(result.redundant);  // false if new

// Specify language
ns.addPhrase('cancel_order', '取消我的订单', 'zh');
```

## Correct a wrong classification

```typescript
// "stop shipment" classified as cancel_order, should be track_order
ns.correct('stop shipment', 'cancel_order', 'track_order');

// Teach from a miss (engine returned nothing)
ns.correct('where did my parcel go', null, 'track_order');
```

## TypeScript types

```typescript
import type { ResolveResult, IntentMatch, PhraseResult } from 'microresolve';

const result: ResolveResult = ns.resolve('cancel my order');
// ResolveResult = { intents: IntentMatch[]; disposition: string }
// IntentMatch   = { id: string; score: number; confidence: number; band: string }

const phraseResult: PhraseResult = ns.addPhrase('cancel_order', 'abort order');
// PhraseResult = { added: boolean; redundant: boolean; warning?: string }
```

## Vercel AI SDK integration

```typescript
import { MicroResolve } from 'microresolve';
import { generateText } from 'ai';

const engine = new MicroResolve({ dataDir: '~/.local/share/microresolve' });
const ns = engine.namespace('support');

async function handleMessage(message: string) {
  const result = ns.resolve(message);

  if (result.disposition === 'Confident') {
    // High confidence — handle directly
    return handleIntent(result.intents[0].id, message);
  }

  // LowConfidence or NoMatch — fall back to LLM
  return generateText({ model: /* ... */, prompt: message });
}
```
